Define your crawler job in TOML format. See documentation for examples.

TOML Format Reference

# Required Fields
id = "my-crawler"              # Unique identifier
name = "My Crawler"            # Human-readable name
description = "Description"    # Job description
source_type = "web"            # Source type (required for crawler jobs)

# Job Type (optional, defaults to "user")
job_type = "user"              # "user" (editable) or "system" (readonly)

# Crawler Configuration
start_urls = ["https://example.com"]
include_patterns = ["^/articles/"]  # Regex patterns to include
exclude_patterns = ["login", "admin"]  # Patterns to exclude
max_depth = 3                  # Maximum crawl depth
max_pages = 100                # Maximum pages to crawl
concurrency = 5                # Concurrent workers
follow_links = true            # Follow discovered links

# Optional Fields
schedule = "0 */6 * * *"       # Cron expression (empty = manual only)
timeout = "30m"                # Duration: 10m, 1h, 30s
enabled = true                 # Enable/disable job
auto_start = false             # Auto-start on application startup
